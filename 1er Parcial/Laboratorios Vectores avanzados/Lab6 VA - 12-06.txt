int buscarMaximo(int vec[], int len)
{
    int posMax = 0;

    for (int i = 1; i < len; i++)
    {
        if (vec[i] > vec[posMax])
        {
            posMax = i;
        }
    }
    return posMax;
}

int buscarMinimo(int vec[], int len)
{
    int posMinimo = 0;

    for (int i = 1; i < len; i++)
    {
        if (vec[i] < vec[posMinimo])
        {
            posMinimo = i;
        }
    }
    return posMinimo;
}

int calcularPromedio(int vec[], int len)
{
    int sum = 0;

    if (len != 0)
    {
        for (int i = 0; i < len; i++)
        {
            sum += vec[i];
        }
        return (sum / len);
    }
    else
        return 0;
}

int sumatoria(int vec[], int len)
{
    int sum = 0;
    for (int i = 0; i < len; i++)
    {
        sum += vec[i];
    }
    return sum;
}
