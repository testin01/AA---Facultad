/*
 *  Sumatoria de todas las notas del primer parcial
 */
#include <iostream>
#include <stdio.h>
#include "principal.hpp"
using namespace std;

struct Alumno
{
    int legajo;
    int dni;
    int nota1;
    int nota2;
};

int main()
{
    FILE *K1151;
    K1151 = fopen("K1151.dat", "rb");

    Alumno raux;
    fread(&raux, sizeof(struct Alumno), 1, K1151);

    int sumatoria = 0;
    while (!feof(K1151))
    {
        sumatoria += (raux.nota1 + raux.nota2);

        fread(&raux, sizeof(struct Alumno), 1, K1151);
    }

    fclose(K1151);
    return 0;
}
