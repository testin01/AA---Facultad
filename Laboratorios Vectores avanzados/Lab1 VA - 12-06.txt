int buscarMinimo(int vec[], int len, int &posMinimo)
{
    posMinimo = 0;

    for (int i = 1; i < len; i++)
    {
        if (vec[i] < vec[posMinimo])
        {
            posMinimo = i;
        }
    }
    return posMinimo;
}
