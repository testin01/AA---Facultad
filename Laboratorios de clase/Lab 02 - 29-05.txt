#include <iostream>
using namespace std;

void inicializarVector(int vector[], int cantPos, int valorInit);
void cargaMasiva(int vector[], int &len, int cantPos);
void promediarVector(int vector[], int &sum, int cantPos, int len);
void mostrarVector(int vector[], int len);

int main()
{
    // Paso 1: Definir el vector
    int vec[5];
    int len = 0;
    int prom = 0;

    // Paso 2: Inicializar
    inicializarVector(vec, 5, 0);
    // Paso 3: Cargar
    cargaMasiva(vec, len, 5);
    // Paso 4: Promediar el vector
    promediarVector(vec, prom, 5, len);
    // Paso 5: Mostrar el result. y el vector
    mostrarVector(vec, len);
    cout << "El promedio es: " << prom << endl;
    return 0;
}

void inicializarVector(int vector[], int cantPos, int valorInit)
{
    for (int i = 0; i < cantPos; i++)
    {
        vector[i] = valorInit;
    }
}

void cargaMasiva(int vector[], int &len, int cantPos)
{
    for (int i = 0; i < cantPos; i++)
    {
        cout << "Ingresar un valor: " << endl;
        cin >> vector[i];
        len++;
    }
}

void promediarVector(int vector[], int &promedio, int cantPos, int len)
{
    int sum = 0;
    for (int i = 0; i < cantPos; i++)
    {
        sum += vector[i];
    }
    promedio = sum/len;
}

void mostrarVector(int vector[], int len)
{
    for (int i = 0; i < len; i++)
    {
        cout << "El valor de v[" << i << "] es: " << vector[i] << endl;
    }
}
